label rotate_vector

// --------------------

// -- out_x = vec_x - vec_y X gyro_z + vec_z X gyro_y
// -- out_y = vec_y - vec_z X gyro_x + vec_x X gyro_z
// -- out_z = vec_z - vec_x X gyro_y + vec_y X gyro_x

// ~~~ X ~~~ -- out_x = vec_x - vec_y X gyro_z + vec_z X gyro_y
LD $GYRO_Z
mov ram A
NEG_A
LD $VECTOR_Y
mov ram X
LD $VECTOR_X
mov ram B
MULADD
mov res B

LD $GYRO_Y
mov ram A
LD $VECTOR_Z
mov ram X
MULADD
LD $VECTOR_NEXT_X
mov res ram


// ~~~ Y ~~~ -- out_y = vec_y - vec_z X gyro_x + vec_x X gyro_z
LD $GYRO_X
mov ram A
NEG_A
LD $VECTOR_Z
mov ram x
LD $VECTOR_Y
mov ram B
MULADD
mov res B

LD $GYRO_Z
mov ram A
LD $VECTOR_X
mov ram X
MULADD
LD $VECTOR_NEXT_Y
mov res ram


// ~~~ Z ~~~ -- out_z = vec_z - vec_x X gyro_y + vec_y X gyro_x
LD $GYRO_Y
mov ram A
NEG_A
LD $VECTOR_X
mov ram X
LD $VECTOR_Z
mov ram B
MULADD
mov res B

LD $GYRO_X
mov ram A
LD $VECTOR_Y
mov ram X
MULADD
LD $VECTOR_NEXT_Z
mov res ram

HALT
